<script>

let game = { hp:100,

maxHp:100,

morality:0, influence:0, starter:null,

inventory:{'Potion':3,'Elixir':1}, currentMonstra:null, enemyMonstra:null

};


const monstraRoster = []; for(let i=1;i<=151;i++){ monstraRoster.push({

id:i, name:"Monstra"+i,

hp:30+Math.floor(Math.random()*70), attack:10+Math.floor(Math.random()*20), defense:5+Math.floor(Math.random()*15), speed:5+Math.floor(Math.random()*15)

});

}


const canvas=document.getElementById('screen'); const ctx=canvas.getContext('2d');


function drawHUD(){ ctx.fillStyle="

306230"; ctx.fillRect(0,0,320,240); if(game.currentMonstra){ ctx.fillStyle="
081820"; ctx.fillRect(20,180,100,10);

let hpWidth = 100 * game.currentMonstra.hp/game.currentMonstra.maxHp; ctx.fillStyle="

Ubbc0f";

ctx.fillRect(20,180,hpWidth,10); ctx.fillStyle="

081820"; ctx.fillText(game.currentMonstra.name+"

HP:"+game.currentMonstra.hp+"/"+game.currentMonstra.maxHp,20,1U5);

}

if(game.enemyMonstra){ ctx.fillStyle="

081820"; ctx.fillRect(200,40,100,10);

let hpWidth = 100 * game.enemyMonstra.hp/game.enemyMonstra.maxHp; ctx.fillStyle="

Ubbc0f";

ctx.fillRect(200,40,hpWidth,10);

ctx.fillStyle="

081820"; ctx.fillText(game.enemyMonstra.name+"

HP:"+game.enemyMonstra.hp+"/"+game.enemyMonstra.maxHp,200,55);

}

}


function startBattle(){ game.currentMonstra = monstraRoster[0];

game.currentMonstra.maxHp = game.currentMonstra.hp; game.enemyMonstra = monstraRoster[Math.floor(Math.random()*151)]; game.enemyMonstra.maxHp = game.enemyMonstra.hp;

document.getElementById("ui").innerHTML = "A wild "+game.enemyMonstra.name+" appeared!";

drawHUD(); showControls();

}


function attack(){

let dmg = game.currentMonstra.attack - game.enemyMonstra.defense/2; dmg = Math.max(1,Math.floor(dmg));

game.enemyMonstra.hp -= dmg; if(game.enemyMonstra.hp<=0){ game.enemyMonstra.hp=0;

document.getElementById("ui").innerHTML="You defeated "+game.enemyMonstra.name+"!";

drawHUD(); return;

}

let enemyDmg = game.enemyMonstra.attack - game.currentMonstra.defense/2; enemyDmg = Math.max(1,Math.floor(enemyDmg));

game.currentMonstra.hp -= enemyDmg; if(game.currentMonstra.hp<=0) game.currentMonstra.hp=0;

document.getElementById("ui").innerHTML="You hit "+game.enemyMonstra.name+" for "+dmg+"!<br>"+

game.enemyMonstra.name+" hits back for "+enemyDmg+"!"; drawHUD();

}


function useItem(item){

if(game.inventory[item] && game.inventory[item]>0){ game.currentMonstra.hp += 30; if(game.currentMonstra.hp>game.currentMonstra.maxHp)

game.currentMonstra.hp=game.currentMonstra.maxHp; game.inventory[item]--;

document.getElementById("ui").innerHTML="Used "+item+"! HP restored."; drawHUD();

}

}


function showControls(){

const c = document.getElementById("controls"); c.innerHTML = "";

["Attack","Potion","Open Shop"].forEach(action=>{ let b=document.createElement("button"); b.innerText=action;

b.onclick = ()=>{ if(action==="Attack") attack();

else if(action==="Potion") useItem("Potion"); else if(action==="Open Shop") openShop();

};

c.appendChild(b);

});

}


function openShop(){ document.getElementById("ui").innerHTML="Welcome to the shop!"; const c = document.getElementById("controls");

c.innerHTML="";

["Potion (+3)","Elixir (+1)"].forEach(item=>{ let b = document.createElement("button"); b.innerText=item;

b.onclick = ()=>{

let key = item.split(" ")[0];

game.inventory[key] = (game.inventory[key]||0)+parseInt(item.match(/\d+/)[0]); document.getElementById("ui").innerHTML="Bought "+item;

};

c.appendChild(b);

});

}


startBattle();
